```html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Title</title>
  <script src="http://d3js.org/d3.v3.min.js" charset="utf-8"></script>
  <style>
    .axis path,
    .axis line{
      fill: none;
      stroke: black;
      shape-rendering: crispEdges;
    }

    .axis text {
      font-family: sans-serif;
      font-size: 11px;
    }
  </style>
</head>
<body>

</body>
<script>
  var width = 400, height = 400;
  var dataset = [100, 120, 140, 160, 180, 200];
  var x_dataset = [1 , 2 , 3 , 4 , 5 , 6 , 7];
  var rect_x = [30 , 60 , 90 , 120 , 150 , 180 , 210];
  var recwidth = 30;
  var svg = d3.select("body")
    .append("svg")
    .attr("width", width)
    .attr("height", height);

  svg.selectAll("rect")
    .data(dataset)
    .enter()
    .append("rect")
    .attr("x",function(d,i){return i * recwidth + 20})
    .attr("y",function (d){return height - d})
    .attr("width" , recwidth - 10)
    .attr("height",function (d){return d})
    .attr("fill","steelblue")
    .attr("transform","translate(30,-50)")

  var x_scale = d3.scale.linear()
  .domain([0 , d3.max(x_dataset)]) //坐标轴上显示的数
  .range([0 , d3.max(rect_x)]) //在整个画布的比例大小

  var y_scale = d3.scale.linear()
  .domain([0,d3.max(dataset)])
  .range([d3.max(dataset) , 0])

  x_axis = d3.svg.axis().scale(x_scale).orient("bottom").ticks(6)
  y_axis = d3.svg.axis().scale(y_scale).orient("left")

  svg.append("g")
  .attr("class","axis")
  .call(x_axis)
  .attr("transform" , "translate(30,350)")

  svg.append("g")
  .attr("transform" , "translate(30 , 150)")
  .attr("class" , "axis")
  .call(y_axis)

  d3.select("body")
  .transition()
  .duration(2000)
  .style("background-color" , "lightblue")

</script>
</html>
```
